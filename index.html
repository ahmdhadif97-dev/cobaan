"smooth" : "auto", }); }; // center on "0001" at mount useEffect(() => { scrollToValue(1, false); }, []); // update displayed number on scroll const handleScroll = () => { if (!containerRef.current) return; const center = containerRef.current.scrollTop + containerRef.current.clientHeight / 2; const index = Math.round(center / itemHeight) - 1; const newValue = Math.min(totalNumbers, Math.max(1, index + 1)); setDisplayed(newValue); }; return (
{/* Scrollable number list */}
