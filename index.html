this.searchInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') { const value = parseInt(this.searchInput.value); if (value >= 1 && value <= 300) { const targetIndex = value - 1; const scrollTop = (targetIndex * this.itemHeight) - (this.pickerContainer.clientHeight / 2 - this.itemHeight / 2); // ✅ set dn immediately so it doesn’t flash one less this.dn = value; this.updateHighlight(); // smooth scroll this.pickerContainer.scrollTo({ top: scrollTop, behavior: 'smooth' }); } } });
