this.searchInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') { const value = parseInt(this.searchInput.value); if (value >= 1 && value <= 300) { const targetIndex = value - 1; const scrollTop = (targetIndex * this.itemHeight) - (this.pickerContainer.clientHeight / 2 - this.itemHeight / 2); // ✅ Manually set dn to the searched value this.dn = value; this.updateHighlight(); // ✅ Smooth scroll without "jumping back one" this.pickerContainer.scrollTo({ top: scrollTop, behavior: 'smooth' }); } } });
